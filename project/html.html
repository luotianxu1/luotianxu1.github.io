                        # 单行溢出显示省略号

                        demo
                        <div class="content ellipsis">单行溢出显示省略号单行溢出显示省略号</div>

    width: 50px;
    height: 40px;
    background-color: #dae9f9;
}

.ellipsis {

    text-overflow: ellipsis;
    white-space: nowrap;
}

                        # 多行溢出显示省略号

                        demo
                        <div class="content ellipsis">

</div>

                        .content {
    width: 150px;
    height: 32px;
    line-height: 16px;
    background-color: #dae9f9;
}

.ellipsis {
    overflow: hidden;
    text-overflow: ellipsis;
    display: -webkit-box;
    -webkit-line-clamp: 2;
    -webkit-box-orient: vertical;
}
