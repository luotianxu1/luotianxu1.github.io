                              export function patch(oldVnode, vnode) {
    // й»ҳи®ӨеҲқе§ӢеҢ–ж—¶пјҢжҳҜзӣҙжҺҘз”ЁиҷҡжӢҹиҠӮзӮ№з©ҝд»¶еӨ„зңҹе®һиҠӮзӮ№ жӣҝжҚўжҺүиҖҒиҠӮзӮ№
    if (oldVnode.nodeType == 1) {
        // з”Ёvnode  жқҘз”ҹжҲҗзңҹе®һdom жӣҝжҚўеҺҹжң¬зҡ„domе…ғзҙ 
        const parentElm = oldVnode.parentNode // жүҫеҲ°д»–зҡ„зҲ¶дәІ
        let elm = createElm(vnode) //ж №жҚ®иҷҡжӢҹиҠӮзӮ№ еҲӣе»әе…ғзҙ 
        parentElm.insertBefore(elm, oldVnode.nextSibling)

        parentElm.removeChild(oldVnode)

        return elm
    } else {
        // еңЁжӣҙж–°зҡ„ж—¶еҖҷпјҢжӢҝиҖҒзҡ„иҷҡжӢҹиҠӮзӮ№е’Ңж–°зҡ„иҷҡжӢҹиҠӮзӮ№еҒҡеҜ№жҜ”пјҢе°ҶдёҚеҗҢзҡ„ең°ж–№жӣҙж–°зңҹе®һзҡ„dom
        // 1гҖҒжҜ”иҫғдёӨдёӘе…ғзҙ зҡ„ж ҮзӯҫпјҢж ҮзӯҫдёҚдёҖж ·зӣҙжҺҘжӣҝжҚўжҺүеҚіеҸҜ
        if (oldVnode.tag !== vnode.tag) {
            return oldVnode.el.parentNode.replaceChild(
                createElm(vnode),
                oldVnode.el
            )
        }

        // 2гҖҒжңүеҸҜиғҪжҳҜж ҮзӯҫдёҖж · <div>1</div> <div>2</div>
        // ж–Үжң¬иҠӮзӮ№зҡ„иҷҡжӢҹиҠӮзӮ№tagйғҪжҳҜundefined
        // ж–Үжң¬зҡ„жҜ”еҜ№
        if (vnode.tag == undefined) {
            // ж–°иҖҒйғҪжҳҜж–Үжң¬
            if (oldVnode.text !== vnode.text) {
                return (el.textContent = vnode.text)
            }
        }

        // ж ҮзӯҫдёҖж ·зӣҙжҺҘеӨҚз”ЁеҚіеҸҜ
        let el = (vnode.el = oldVnode.el) // еӨҚз”ЁиҖҒиҠӮзӮ№

        // 3гҖҒж ҮзӯҫдёҖж · е№¶дё”йңҖиҰҒејҖе§ӢжҜ”еҜ№ж ҮзӯҫеұһжҖ§е’Ңе„ҝеӯҗ
        // жӣҙж–°еұһжҖ§пјҢз”Ёж–°зҡ„иҷҡжӢҹиҠӮзӮ№зҡ„еұһжҖ§е’ҢиҖҒзҡ„жҜ”иҫғпјҢеҺ»жӣҙж–°иҠӮзӮ№
        updateProperties(vnode, oldVnode.data)

        // е„ҝеӯҗжҜ”иҫғж°ӣеӣҙдёҖдёӢеҮ з§Қжғ…еҶө
        let oldChildren = oldVnode.children || []
        let newChildren = vnode.children || []
        if (oldChildren.length > 0 && newChildren.length > 0) {
            // еҸҢж–№йғҪжңүе„ҝеӯҗ
            // vueз”ЁдәҶеҸҢжҢҮй’Ҳзҡ„ж–№ејҸ жқҘжҜ”еҜ№
            patchChildren(el, oldChildren, newChildren)
        } else if (newChildren.length > 0) {
            // иҖҒзҡ„жІЎе„ҝеӯҗ дҪҶжҳҜж–°зҡ„жңүе„ҝеӯҗ
            for (let i = 0; i < newChildren.length; i++) {
                let child = createElm(newChildren[i])
                el.appendChild(child) // еҫӘзҺҜеҲӣе»әж–°иҠӮзӮ№
            }
        } else if (oldChildren.length > 0) {
            // иҖҒзҡ„жңүе„ҝеӯҗ ж–°зҡ„жІЎжңүе„ҝеӯҗ
            el.innerHTML = `` // зӣҙжҺҘеҲ йҷӨиҖҒиҠӮзӮ№
        }
    }
}

export function createElm(vnode) {
    let { tag, data, children, text, vm } = vnode
    if (typeof tag === 'string') {
        // е…ғзҙ 
        vnode.el = document.createElement(tag) // иҷҡжӢҹиҠӮзӮ№дјҡжңүдёҖдёӘelеұһжҖ§ еҜ№еә”зңҹе®һиҠӮзӮ№

        // еҸӘжңүе…ғзҙ жүҚжңүеұһжҖ§
        updateProperties(vnode)

        children.forEach((child) => {
            vnode.el.appendChild(createElm(child))
        })
    } else {
        vnode.el = document.createTextNode(text)
    }
    return vnode.el
}

                              function updateProperties(vnode, oldProps = {}) {
    let newProps = vnode.data || {} // ж–°зҡ„еұһжҖ§
    let el = vnode.el

    // ж ·ејҸеӨ„зҗҶ иҖҒзҡ„style={color:red} ж–°зҡ„ style={background: red}
    let newStyle = newProps.style || {}
    let oldStyle = oldProps.style || {}
    // иҖҒзҡ„ж ·ејҸдёӯжңү ж–°зҡ„жІЎжңү еҲ йҷӨиҖҒзҡ„ж ·ејҸ
    for (let key in oldStyle) {
        if (!newStyle[key]) {
            // ж–°зҡ„йҮҢйқўдёҚеӯҳеңЁиҝҷдёӘж ·ејҸ
            el.style[key] = ''
        }
    }

    // иҖҒзҡ„жңүж–°зҡ„жІЎжңү йңҖиҰҒеҲ йҷӨеұһжҖ§
    for (let key in oldProps) {
        if (!newProps[key]) {
            el.removeAttribute(key) // з§»йҷӨзңҹе®һdomзҡ„еұһжҖ§
        }
    }

    // ж–°зҡ„жңү йӮЈе°ұзӣҙжҺҘз”Ёж–°зҡ„еҺ»еҒҡжӣҙж–°еҚіеҸҜ
    for (let key in newProps) {
        if (key == 'style') {
            for (let styleName in newProps.style) {
                el.style[styleName] = newProps.style[styleName]
            }
        } else if (key == 'class') {
            el.className = newProps.class
        } else {
            el.setAttribute(key, newProps[key])
        }
    }
}

                              function isSameVnode(oldVnode, newVnode) {
    return oldVnode.tag == newVnode.tag && oldVnode.key == newVnode.key
}

function patchChildren(el, oldChildren, newChildren) {
    // DOMдёӯж“ҚдҪңжңүеҫҲеӨҡеёёи§Ғзҡ„йҖ»иҫ‘ жҠҠиҠӮзӮ№жҸ’е…ҘеҲ°еҪ“еүҚе„ҝеӯҗзҡ„еӨҙйғЁгҖҒе°ҫйғЁгҖҒе„ҝеӯҗеҖ’еҸҷжӯЈеәҸ

    let oldStartIndex = 0 // иҖҒзҡ„зҙўеј•
    let oldStartVnode = oldChildren[0] // иҖҒзҡ„зҙўеј•жҢҮеҗ‘зҡ„иҠӮзӮ№
    let oldEndIndex = oldChildren.length - 1
    let oldEndVnode = oldChildren[oldEndIndex]
    let newStartIndex = 0
    let newStartVnode = newChildren[0]
    let newEndIndex = newChildren.length - 1
    let newEndVnode = newChildren[newEndIndex]

    const makeIndexByKey = (children) => {
        return children.reduce((memo, current, index) => {
            if (current.key) {
                memo[current.key] = index
            }
            return memo
        }, {})
    }
    // еҲӣе»әжҳ е°„иЎЁ
    const keysMap = makeIndexByKey(oldChildren)

    // еҒҡдёҖдёӘеҫӘзҺҜеҗҢж—¶еҫӘзҺҜиҖҒзҡ„е’Ңж–°зҡ„пјҢе“ӘдёӘе…Ҳз»“жқҹ еҫӘзҺҜе°ұеҒңжӯўпјҢе°ҶеӨҡдҪҷзҡ„еҲ йҷӨжҲ–иҖ…ж·»еҠ иҝӣеҺ»
    while (oldStartIndex <= oldEndIndex && newStartIndex <= newEndIndex) {
        // еӨҙеӨҙжҜ”иҫғ е°ҫе°ҫжҜ”иҫғ еӨҙе°ҫжҜ”иҫғ е°ҫеӨҙжҜ”иҫғ
        // дјҳеҢ–дәҶ еҗ‘еҗҺж·»еҠ пјҢ еҗ‘еүҚж·»еҠ пјҢе°ҫе·ҙз§»еҠЁеҲ°еӨҙйғЁпјҢеӨҙйғЁз§»еҠЁеҲ°е°ҫйғЁ пјҢеҸҚиҪ¬
        if (!oldStartVnode) {
            // е·Із»Ҹиў«з§»еҠЁиө°дәҶ
            oldStartVnode = oldChildren[++oldStartIndex]
        } else if (!oldEndVnode) {
            oldEndVnode = oldChildren[--oldEndIndex]
        }

        // еҗҢж—¶еҫӘзҺҜж–°зҡ„иҠӮзӮ№е’Ң иҖҒзҡ„иҠӮзӮ№пјҢжңүдёҖж–№еҫӘзҺҜе®ҢжҜ•е°ұз»“жқҹдәҶ
        if (isSameVnode(oldStartVnode, newStartVnode)) {
            // еӨҙеӨҙжҜ”иҫғпјҢеҸ‘зҺ°ж ҮзӯҫдёҖиҮҙпјҢ
            patch(oldStartVnode, newStartVnode)
            oldStartVnode = oldChildren[++oldStartIndex]
            newStartVnode = newChildren[++newStartIndex]
        } else if (isSameVnode(oldEndVnode, newEndVnode)) {
            // д»Һе°ҫйғЁејҖе§ӢжҜ”иҫғ
            patch(oldEndVnode, newEndVnode)
            oldEndVnode = oldChildren[--oldEndIndex]
            newEndVnode = newChildren[--newEndIndex]
        }

        // еӨҙе°ҫжҜ”иҫғ  =гҖӢ reverse
        else if (isSameVnode(oldStartVnode, newEndVnode)) {
            patch(oldStartVnode, newEndVnode)
            el.insertBefore(oldStartVnode.el, oldEndVnode.el.nextSibling) // з§»еҠЁиҖҒзҡ„е…ғзҙ пјҢиҖҒзҡ„е…ғзҙ е°ұиў«з§»еҠЁиө°дәҶпјҢдёҚз”ЁеҲ йҷӨ
            oldStartVnode = oldChildren[++oldStartIndex]
            newEndVnode = newChildren[--newEndIndex]
        } else if (isSameVnode(oldEndVnode, newStartVnode)) {
            // е°ҫеӨҙжҜ”иҫғ
            patch(oldEndVnode, newStartVnode)
            el.insertBefore(oldEndVnode.el, oldStartVnode.el)
            oldEndVnode = oldChildren[--oldEndIndex]
            newStartVnode = newChildren[++newStartIndex]
        } else {
            // д№ұеәҸжҜ”еҜ№   ж ёеҝғdiff
            // 1.йңҖиҰҒж №жҚ®keyе’Ң еҜ№еә”зҡ„зҙўеј•е°ҶиҖҒзҡ„еҶ…е®№з”ҹжҲҗзЁӢжҳ е°„иЎЁ
            let moveIndex = keysMap[newStartVnode.key] // з”Ёж–°зҡ„еҺ»иҖҒзҡ„дёӯжҹҘжүҫ
            // дёҚйңҖиҰҒз§»еҠЁиҜҙжҳҺжІЎжңүkeyеӨҚз”Ёзҡ„
            if (moveIndex == undefined) {
                // еҰӮжһңдёҚиғҪеӨҚз”ЁзӣҙжҺҘеҲӣе»әж–°зҡ„жҸ’е…ҘеҲ°иҖҒзҡ„иҠӮзӮ№ејҖеӨҙеӨ„
                el.insertBefore(createElm(newStartVnode), oldStartVnode.el)
            } else {
                let moveNode = oldChildren[moveIndex] // иҖҒзҡ„иҷҡжӢҹиҠӮзӮ№
                oldChildren[moveIndex] = null // жӯӨиҠӮзӮ№е·Із»Ҹиў«з§»еҠЁиө°дәҶ
                el.insertBefore(moveNode.el, oldStartVnode.el)
                patch(moveNode, newStartVnode) // жҜ”иҫғдёӨдёӘиҠӮзӮ№зҡ„еұһжҖ§
            }
            newStartVnode = newChildren[++newStartIndex]
        }
    }
    //  еҰӮжһңиҖҒзҡ„еӨҡ е°ҶиҖҒиҠӮзӮ№еҲ йҷӨ пјҢ дҪҶжҳҜеҸҜиғҪйҮҢйқўжңүnull зҡ„жғ…еҶө
    if (oldStartIndex <= oldEndIndex) {
        for (let i = oldStartIndex; i <= oldEndIndex; i++) {
            if (oldChildren[i] !== null) el.removeChild(oldChildren[i].el)
        }
    }
    // еҰӮжһңз”ЁжҲ·иҝҪеҠ дәҶдёҖдёӘжҖҺд№ҲеҠһпјҹ

    // иҝҷйҮҢжҳҜжІЎжңүжҜ”еҜ№е®Ңзҡ„
    if (newStartIndex <= newEndIndex) {
        for (let i = newStartIndex; i <= newEndIndex; i++) {
            // el.appendChild(createElm(newChildren[i]))
            // insertBeforeж–№жі• д»–еҸҜд»ҘappendChildеҠҹиғҪ insertBefore(иҠӮзӮ№,null)  dom api

            //  зңӢдёҖдёӢдёәжҢҮй’Ҳзҡ„дёӢдёҖдёӘе…ғзҙ жҳҜеҗҰеӯҳеңЁ
            let anchor =
                newChildren[newEndIndex + 1] == null
                    ? null
                    : newChildren[newEndIndex + 1].el
            el.insertBefore(createElm(newChildren[i]), anchor)
        }
    }
}

export function lifecycleMixin(Vue) {
    Vue.prototype._update = function (vnode) {
        const vm = this
        // иҝҷйҮҢйңҖиҰҒеҢәеҲҶдёӢеҲ°еә•жҳҜйҰ–ж¬ЎжёІжҹ“иҝҳжҳҜжӣҙж–°
        const prevVnode = vm._vnode
        if (!prevVnode) {
            // з”Ёж–°зҡ„еҲӣе»әзҡ„е…ғзҙ пјҢжӣҝжҚўиҖҒзҡ„vm.$el
            vm.$el = patch(vm.$el, vnode)
        } else {
            vm.$el = patch(prevVnode, vnode)
        }
        vm._vnode = vnode
    }
}
